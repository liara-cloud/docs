نمونه اول:

<div dir='ltr'>

        {`const Koa = require('koa');
const app = new Koa();
app.proxy = true;

app.use(async ctx => {
  const ip = ctx.request.ip;
  ctx.body = \`Client IP: \${ip}\`;
});`}
    </Highlight>
</div>

});`}
