        "Authorization": f"Bearer {settings.LIARA_API_TOKEN}",
        "Content-Type": "application/json",

    }

    data = {

        "subject": subject,
        "to": to,
        "from": settings.MAIL_FROM,
        "attachments": []
    }

    try:
        response = requests.post(url, json=data, headers=headers)
