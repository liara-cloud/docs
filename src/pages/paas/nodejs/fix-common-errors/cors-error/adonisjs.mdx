# رفع خطای CORS در فریم‌ورک AdonisJS
<hr className="mb-2" />

برای رفع این خطا در فریم‌ورک AdonisJS، کافیست تا وارد فایل <Important>config/cors.js</Important> یا <Important>config/cors.ts</Important> شوید و مقادیر <Important>origin</Important> و <Important>methods</Important> را در این فایل، مانند شکل زیر تنظیم کنید:
<div className="h-2" />
<div dir='ltr'>
    <Highlight className="bash">
        {`'use strict'

module.exports = {
  origin: '*',
  methods: ['GET', 'PUT', 'PATCH', 'POST', 'DELETE', 'HEAD'],
  headers: true,
  exposeHeaders: false,
  credentials: false,
  maxAge: 90
}`}
    </Highlight>
</div>
<div className="h-2" />
<Alert variant='success'>
<p>

</p>
